## Install

```bash
npm install --save react-image-magnify-lib
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-image-magnify-lib'
import 'react-image-magnify-lib/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
